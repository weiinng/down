		<script type="text/javascript">
			$(function(){     //当页面标签加载完毕之后加载jquery 
				
				var menu = $('.menu');    //获取菜单
				var menu_back = $('.menu_back');
				var totop = $('.totop')
				
				$(window).scroll(function(){
					var inum = $(document).scrollTop();  //获取滚动的值					
					if(inum > 200){   //如果滚动值大于200则执行以下操作
						menu.css({     //修改menu的css属性
							'position':'fixed',            //脱离文档流
							'left':'50%',               //设置left的值为 50%
							'marginLeft':'-480px',     //在减去它本身的 一般的 px 值
							'top':0,      //这是top为零
						});
						menu_back.show()  //显示这个元素
						
					}
					else{      //如果 meun 的值不是大于200 则执行以下操作
						menu.css({      //更改 meun的css 属性
							'position':'static',    //还原定位
							'marginLeft':'auto',     //将次属性改为 居中
						})
					}
					
					
					if(inum>400){    
						totop.fadeIn()
					}
					else{
						totop.fadeOut()
					}
					
					
				})
				totop.click(function(){   //赋予totop一个点击事件
					$('html,body').animate({   //给html或者body的标签 赋予一个动画
						'scrollTop':'0',    //拉动滚动条这个值会有变化如果 点击这个时间这个值会为0
					})
				})
				
				
				
			})
