class A:
    __money=1
    def __init__(self):
        self.name='父亲'
        self.age=18
    def t(self):
        print(self.name)
        return self.name
class B(A):
    def t(self):
        self.name='儿子'
        print(self.name)
    def t2(self):
        print(self.name)   
        A.__init__(self)   #将self.name改为父亲
        print(self.name)
    def t3(self):
        n=A.t(self)
        print(n)
        print(self.age)
b=B()
b.t()
b.t2()
b.t3()
