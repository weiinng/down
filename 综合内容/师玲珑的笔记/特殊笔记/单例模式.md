#定义一个单例模式的装饰器
def singleton(cls):
    #定义实例容器
    instances={}
    #定义嵌套方法
    def wrapper(*args,**kwargs):  #装饰器自身性质 内置函数里的参数就是你穿过来的类的实例化参数
        #判断装饰的类是否为单例
        if cls not in instances:
            instances[cls]=cls(*args,**kwargs)
        return instances[cls]
    return wrapper
